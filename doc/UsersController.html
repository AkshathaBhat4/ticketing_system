def create
  @user = User.new(user_params)

  if @user.save
    render json: @user.as_json
  else
    render json: @user.errors.to_json, status: :unprocessable_entity
  end
end

def update
  @user = User.find(params[:id])
  if @user.update(user_params)
    render json: @user.as_json
  else
    render json: @user.errors.to_json, status: :unprocessable_entity
  end
end
